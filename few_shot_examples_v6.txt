================================================================
FEW-SHOT EXAMPLES (v6)
Each example shows: input rows for ONE benefit + the two text fields
the LLM should produce. Python handles all the IDs and names.
================================================================

--- EXAMPLE 1: Simple copay ---
CONTEXT:
  benefit: 900 - Doctor Office Visits Primary
  coverage: InNetwork, service: N/A
INPUT ROWS:
  [Primary Care Physician Services (7a) - Medicare] Is there a copayment? = Yes
  [Primary Care Physician Services (7a) - Medicare] Copayment amount = $0
  [Primary Care Physician Services (7a) - Medicare] Is there a coinsurance? = No
OUTPUT:
  {"benefitdesc": "<b>$0</b> copay per visit (Medicare Covered)", "tinyDescription": "$0"}

--- EXAMPLE 2: Copay with treatment limit and periodicity ---
CONTEXT:
  benefit: 1900 - Acupuncture
  coverage: InNetwork, service: N/A
INPUT ROWS:
  [Acupuncture - Number of Treatments (13a) - Non-Medicare] Is there a copayment? = Yes
  [Acupuncture - Number of Treatments (13a) - Non-Medicare] Copayment amount = $25
  [Acupuncture - Number of Treatments (13a) - Non-Medicare] Indicate limit for Number of Treatments = 20
  [Acupuncture - Number of Treatments (13a) - Non-Medicare] Periodicity = Every Year
OUTPUT:
  {"benefitdesc": "<b>$25</b> copay<br> for 20 treatment(s) every year", "tinyDescription": "$25"}

--- EXAMPLE 3: Min/Max copay range ---
CONTEXT:
  benefit: 1400 - Hearing Services
  coverage: InNetwork, service: Prescription Hearing Aid
INPUT ROWS:
  [Prescription Hearing Aids (18b1) - Non-Medicare] Is there a copayment? = Yes with a min & max
  [Prescription Hearing Aids (18b1) - Non-Medicare] Minimum copayment = $699
  [Prescription Hearing Aids (18b1) - Non-Medicare] Maximum copayment = $999
  [Prescription Hearing Aids (18b1) - Non-Medicare] Indicate quantity for Hearing Aids = 2
  [Prescription Hearing Aids (18b1) - Non-Medicare] Periodicity = Every Year
OUTPUT:
  {"benefitdesc": "<b>$699 - $999</b> copay<br> for 2 hearing aids for every year", "tinyDescription": "$699 - $999"}
