-- Claim 1: Count of corpids with 2+ waves (wave = sequential campaign order per corp)
WITH corp_waves AS (
    SELECT
        corpid,
        campaign_id,
        strategy,
        ROW_NUMBER() OVER (
            PARTITION BY corpid
            ORDER BY campaign_start_date  -- replace with actual date column
        ) AS wave_number
    FROM curatedstore.t_stage_hyperlocal_zip_selection
    GROUP BY corpid, campaign_id, strategy, campaign_start_date
)
SELECT
    campaign_id,
    strategy,
    COUNT(DISTINCT corpid) AS clients_with_2plus_waves,
    47 AS claimed_value,
    CASE WHEN COUNT(DISTINCT corpid) = 47 THEN 'VERIFIED' ELSE 'MISMATCH' END AS status
FROM (
    SELECT corpid, campaign_id, strategy
    FROM corp_waves
    GROUP BY corpid, campaign_id, strategy
    HAVING MAX(wave_number) >= 2
) wave_counts
GROUP BY campaign_id, strategy
ORDER BY campaign_id;


-- Claim 2: Median CPA by wave (trajectory + % drop)
WITH corp_waves AS (
    SELECT
        corpid,
        campaign_id,
        strategy,
        cpa,
        ROW_NUMBER() OVER (
            PARTITION BY corpid
            ORDER BY campaign_start_date  -- replace with actual date column
        ) AS wave_number
    FROM curatedstore.t_stage_hyperlocal_zip_selection
    WHERE cpa IS NOT NULL
),
wave_medians AS (
    SELECT
        campaign_id,
        strategy,
        wave_number,
        PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY cpa) AS median_cpa,
        COUNT(DISTINCT corpid) AS n_clients
    FROM corp_waves
    GROUP BY campaign_id, strategy, wave_number
),
wave_with_lag AS (
    SELECT
        campaign_id,
        strategy,
        wave_number,
        median_cpa,
        n_clients,
        FIRST_VALUE(median_cpa) OVER (
            PARTITION BY campaign_id, strategy
            ORDER BY wave_number
            ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
        ) AS wave1_cpa,
        LAST_VALUE(median_cpa) OVER (
            PARTITION BY campaign_id, strategy
            ORDER BY wave_number
            ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
        ) AS last_wave_cpa
    FROM wave_medians
)
SELECT
    campaign_id,
    strategy,
    wave_number,
    n_clients,
    ROUND(median_cpa, 0)  AS median_cpa,
    ROUND(wave1_cpa, 0)   AS starting_cpa,
    ROUND(last_wave_cpa, 0) AS ending_cpa,
    ROUND(100.0 * (wave1_cpa - last_wave_cpa) / NULLIF(wave1_cpa, 0), 1) AS pct_drop,
    CASE
        WHEN ROUND(100.0 * (wave1_cpa - last_wave_cpa) / NULLIF(wave1_cpa, 0), 0) BETWEEN 60 AND 70
        THEN 'VERIFIED'
        ELSE 'MISMATCH'
    END AS status
FROM wave_with_lag
ORDER BY campaign_id, wave_number;


-- Claim 3: Full CPA trajectory per wave (pivot)
WITH corp_waves AS (
    SELECT
        corpid,
        campaign_id,
        strategy,
        cpa,
        ROW_NUMBER() OVER (
            PARTITION BY corpid
            ORDER BY campaign_start_date  -- replace with actual date column
        ) AS wave_number
    FROM curatedstore.t_stage_hyperlocal_zip_selection
    WHERE cpa IS NOT NULL
),
wave_medians AS (
    SELECT
        campaign_id,
        strategy,
        wave_number,
        PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY cpa) AS median_cpa
    FROM corp_waves
    GROUP BY campaign_id, strategy, wave_number
)
SELECT
    campaign_id,
    strategy,
    MAX(CASE WHEN wave_number = 1 THEN ROUND(median_cpa, 0) END) AS w1_cpa,
    MAX(CASE WHEN wave_number = 2 THEN ROUND(median_cpa, 0) END) AS w2_cpa,
    MAX(CASE WHEN wave_number = 3 THEN ROUND(median_cpa, 0) END) AS w3_cpa,
    MAX(CASE WHEN wave_number = 4 THEN ROUND(median_cpa, 0) END) AS w4_cpa,
    MAX(CASE WHEN wave_number = 5 THEN ROUND(median_cpa, 0) END) AS w5_cpa,
    CASE
        WHEN MAX(CASE WHEN wave_number = 1 THEN ROUND(median_cpa, 0) END) = 500
         AND MAX(CASE WHEN wave_number = 5 THEN ROUND(median_cpa, 0) END) = 178
        THEN 'VERIFIED'
        ELSE 'MISMATCH'
    END AS status
FROM wave_medians
GROUP BY campaign_id, strategy
ORDER BY campaign_id;
